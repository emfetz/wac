    def inspect
      out = "q: \"#{input}\""
      out << " #{options[:podstate]}" if options[:podstate]
      out << " (assuming #{options[:assumption]})" if options[:assumption]
      out << ", a: #{result.datatypes}" if fetched?
      out
    end
